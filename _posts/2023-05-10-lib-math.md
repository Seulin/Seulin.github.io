```python
import math

# Round up
math.ceil(1.2) # 2
math.ceil(-3.2) # -3

# Round down
math.floor(1.8) # 1
int(1.8) # 1
## Difference between int() and math.floor()
math.floor(-3.8) # -4
int(-3.8) # -3

# Round off
round(1.2) # 1
round(0.126, 0) # 0.0
round(0.126, 2) # 0.13
## If a number ends with 5, the last digit to be displayed should be even
round(0.125, 2) # 0.12
round(0.115, 2) # 0.12

# Natural Logarithm
math.log(math.e**2) # 2.0
# Common Logarithm
math.log10(100) # 2.0
math.log2(32) # 5.0

# Infinity
math.inf > math.pow(2, 1000) # True

# Root
math.sqrt(3**6) # 27.0
math.cbrt(3**6) # 9.000000000000002 / due to finite precision

# Greatest Common Divisor
math.gcd(3, 39, 5, 7) # 1
# Leastest Common Multiple
math.lcm(3, 39, 5, 7) # 1365

# Permutation
math.perm(10, 4) # 210
# Combination
math.comb(10, 4) # 5040
# Product
math.prod({5, 3, 10}) # 150

# Factorial
math.factorial(5) # 120

# +divmod (Quotient, Remainder)
divmod(8, 3) # (2, 2)
```
